### 安装node.js
    http://nodejs.org/


### 在命令行执行一下命令
    git clone git@git.mofang.com:zhangzhiwei/mfm.git ~/mfm && cd ~/mfm
    npm remove mfe -g && npm remove mfm -g && npm link .
    cd ~ # done ..

### 开始模块:
    mkdir workspace && cd workspace
    mfm sm test

### 更多细节
    mfm --help

### 更新:
    mfm update
